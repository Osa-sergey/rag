# ══════════════════════════════════════════════════════════════════
#  ACL — Управление доступом к этому файлу
# ══════════════════════════════════════════════════════════════════
#
#  Каждая запись в блоке acl: задаёт одно правило доступа.
#  Правило автоматически привязывается к пути ЭТОГО файла.
#
#  Доступные поля:
#    users:        [список конкретных пользователей]
#    roles:        [список ролей]
#    permissions:  [read, write, delete, create, all]
#    deny:         true/false  (по умолчанию false = разрешение)
#    priority:     смещение от базы 200 (необязательно)
#
#  Приоритет inline-правил по умолчанию = 200, что выше
#  глобальных правил из policy.yaml (обычно 0–100).
#  Если нужно ещё выше — добавьте priority: 10 → итого 210.
# ══════════════════════════════════════════════════════════════════

acl:
  # ── 1. Полный доступ конкретному пользователю ──────────────────
  #    alice — автор документа, может делать всё
  - users: [alice]
    permissions: [all]

  # ── 2. Персональный доступ на чтение и редактирование ──────────
  #    bob и charlie могут читать и редактировать,
  #    но НЕ удалять (delete не указан → не выдаётся)
  - users: [bob, charlie]
    permissions: [read, write]

  # ── 3. Доступ по роли — только чтение ──────────────────────────
  #    Все с ролью "viewer" могут только читать этот документ
  - roles: [viewer]
    permissions: [read]

  # ── 4. Расширенный доступ для роли + конкретного пользователя ──
  #    Роль "architect" получает чтение и запись.
  #    Дополнительно — frank лично может удалять.
  - roles: [architect]
    permissions: [read, write]
  - users: [frank]
    permissions: [read, write, delete]

  # ── 5. Явный ЗАПРЕТ операции для роли ──────────────────────────
  #    Все редакторы (editor) НЕ МОГУТ удалять этот файл,
  #    даже если глобальная политика им это разрешает.
  - roles: [editor]
    permissions: [delete]
    deny: true

  # ── 6. Запрет конкретному пользователю ─────────────────────────
  #    dave не может ни читать, ни редактировать этот документ,
  #    даже если его роль (viewer) глобально разрешает чтение.
  #    Повышенный priority (+5) гарантирует, что это правило
  #    перевесит правило №3 выше (у обоих base = 200,
  #    но у этого итого 205 > 200).
  - users: [dave]
    permissions: [read, write]
    deny: true
    priority: 5

  # ── 7. Временный гостевой доступ ───────────────────────────────
  #    Роль "guest" — только чтение, без возможности
  #    что-либо менять или удалять
  - roles: [guest]
    permissions: [read]
